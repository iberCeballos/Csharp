            {                
                Console.WriteLine("La fecha es correcta");
            }
            else
            {
               Console.WriteLine("La fecha es incorrecta");
            }
            
            years = currenDate.Year - birthday.Year;
            
            if (birthday.Month >= currenDate.Month)
            {                
                years--;

                }
