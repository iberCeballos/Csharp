                Console.WriteLine($"los numeros que usted ingreso con impares y el resultado de la multiplicaci√≤n es:{multiplicacion}");
                
            }

                Console.WriteLine($"los numeros que usted ingreso son pares y el resultado de la suma es:{suma}");

            }






        }
    }
